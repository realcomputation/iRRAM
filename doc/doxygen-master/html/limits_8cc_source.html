45 REAL limit_lip (REAL (*f)(int,const REAL&),
46  int lip_value,
47  bool (*on_domain)(const REAL&),
48  const REAL& x)
49 {
50  if ( on_domain(x) != true ) REITERATE(0);
51 
52  limit_computation env;
53 
54  REAL x_new,lim;
55  sizetype x_error;
56 
57  x_new=x;
58  x_new.geterror(x_error);
59  assert(x_new.value);
60  sizetype_exact(x_new.error);
61 
62  limit_debug("starting limit_lip1");
63 
64  while (1) {
65  try{
66  iRRAM_DEBUG2(2,"trying to compute limit_lip1 with precision %d...\n",state.ACTUAL_STACK.actual_prec);
67  lim=f(env.saved_prec(),x_new);
68  if (lim.error.exponent > env.saved_prec()) {
69  env.inc_step(2);
70  iRRAM_DEBUG2(2,"limit_lip1 too imprecise, increasing precision locally to %d...\n",state.ACTUAL_STACK.actual_prec);
71  } else {
72  iRRAM_DEBUG2(2,"getting result with local error %d*2^(%d)\n",
73  lim.error.mantissa, lim.error.exponent);
74  break;
75  }}
76  catch ( Iteration it){
77  env.inc_step(2);
78  iRRAM_DEBUG2(2,"limit_lip1 failed, increasing precision locally to %d...\n",state.ACTUAL_STACK.actual_prec);
79  } }
80  lim.adderror(sizetype_add_power2(x_error << lip_value, env.saved_prec()));
81  iRRAM_DEBUG2(2,"end of limit_lip1 with error %d*2^(%d)\n"
82  " error of argument: %d*2^(%d)\n",
83  lim.error.mantissa,lim.error.exponent,
84  x_error.mantissa,x_error.exponent);
85  return lim;
86 }

207 REAL lipschitz (REAL (*f)(const REAL&),
208  REAL (*lip_f)(const REAL&),
209  bool (*on_domain)(const REAL&),
210  const REAL& x)
211 {
212  if ( on_domain(x) != true ) REITERATE(0);
213 
214  REAL x_new,lip_result,lip_bound;
215  sizetype x_error;
216 
217  {
218  limit_computation env(0);
219  iRRAM_DEBUG1(2,"starting lipschitz1b ...\n");
220 
221 // for the computation of the Lipschitz bound, we work with
222 // reduced precision:
223  {
224  relaxed rel;
225  lip_bound=lip_f(x);
226  }
227  x_new=x;
228  x_new.geterror(x_error);
229  assert(x_new.value);
230  sizetype_exact(x_new.error);
231 
232 
233  bool try_it=true;
234  while (try_it) {
235  try { try_it=false;
236  lip_result=f(x_new); }
237  catch ( Iteration it) { try_it=true;
238  env.inc_step(2);
239  iRRAM_DEBUG2(2,"limit_lip2 failed, increasing precision locally to %d...\n",state.ACTUAL_STACK.actual_prec);
240  }
241  }
242  }
243 
244  iRRAM_DEBUG2(2,"getting result with local error %d*2^(%d)\n",
245  lip_result.error.mantissa, lip_result.error.exponent);
246  lip_result.adderror((lip_bound.getsize() + lip_bound.geterror()) * x_error);
247  iRRAM_DEBUG2(2,"end of lipschitz_1b with error %d*2^(%d)\n"
248  " for argument with error %d*2^(%d)\n",
249  lip_result.error.mantissa,lip_result.error.exponent,
250  x_error.mantissa,x_error.exponent);
251  return lip_result;
252 }

390 REAL limit_hint (REAL (*f)(int, const REAL&),
391  int hint,
392  const REAL& x)
393 {
394  limit_computation env;
395 
396  REAL lim,limnew;
397  sizetype limnew_error;
398  sizetype lim_error;
399  int hintcopy=hint;
400  int success=0;
401 
402  int element=max(x.geterror().exponent,env.saved_prec());
403 
404  iRRAM_DEBUG1(2,"starting limit_hint1...\n");
405 
406  while (1) {
407  try {
408  iRRAM_DEBUG2(2,"trying to compute limit_hint1 with precicion 2^(%d)...\n",element);
409  limnew=f(element,x);
410  limnew_error = sizetype_add_power2(limnew.geterror(), element);
411  if ( (! success) || sizetype_less(limnew_error,lim_error) ) {
412  lim=limnew;
413  lim_error=limnew_error;
414  iRRAM_DEBUG2(2,"getting result with error %d*2^(%d)\n",
415  lim_error.mantissa, lim_error.exponent);
416  } else {
417  iRRAM_DEBUG1(2,"computation successful, but no improvement\n");
418  hintcopy=2*hintcopy;
419  }
420  success=1;
421  if (element>iRRAM_prec_array[1])break;
422  element=element+hint;
423  }
424  catch ( Iteration it) {
425  if (element>iRRAM_prec_array[1])break;
426  element=element+hintcopy;
427  }}
428  if ( ! success) {
429  iRRAM_DEBUG1(1,"computation of limit_hint1 failed\n");
430  REITERATE(0);
431  }
432  lim.seterror(lim_error);
433  iRRAM_DEBUG2(2,"end of limit_hint1 with error %d*2^(%d)\n",
434  lim_error.mantissa,lim_error.exponent);
435  return lim;
436 }

659 // REAL limit ( FUNCTION<REAL,int> f )
660 // if FUNCTION f defines a normed Cauchy sequence, i.e. |f(i)-x|<= 2^{i}
661 // then limit(f) returns x
662 //********************************************************************************
663 
664 REAL limit (const FUNCTION<REAL,int> & f )
665 {
666  limit_computation env;
667 
668  REAL lim,limnew;
669  sizetype limnew_error;
670  sizetype lim_error;
671 
672  int element=env.saved_prec();
673  int element_step=env.saved_step();
674  int firsttime=2;
675 
676  limit_debug("starting limit_FUNCTION");
677 
678  while (1) {
679  try {
680  iRRAM_DEBUG2(2,"trying to compute limit_FUNCTION with precicion 2^(%d)...\n",element);
681  limnew=f(element);
682  limnew_error = sizetype_add_power2(limnew.geterror(), element);
683  if (firsttime ==2 ) if ( limnew_error.exponent > env.saved_prec(-1)) {
684  iRRAM_DEBUG0(2,{cerr<<"computation not precise enough ("
685  << limnew_error.mantissa <<"*2^"<< limnew_error.exponent
686  <<"), trying normal p-sequence\n";});
687  element_step=1;
688  element=4+iRRAM_prec_array[element_step];
689  firsttime=1;
690  }
691  if ( firsttime != 0 || sizetype_less(limnew_error,lim_error) ) {
692  lim=limnew;
693  lim_error=limnew_error;
694  iRRAM_DEBUG2(2,"getting result with error %d*2^(%d)\n",
695  lim_error.mantissa, lim_error.exponent);
696  } else {
697  iRRAM_DEBUG1(2,"computation successful, but no improvement\n");
698  }
699  firsttime=0;
700  if (element<=env.saved_prec())break;
701  element_step+=4;
702  element=iRRAM_prec_array[element_step];
703  }
704  catch ( Iteration it) {
705  if ( firsttime==0) {
706  iRRAM_DEBUG1(2,"computation failed, using best success\n");
707  break;
708  } else
709  if ( firsttime==2) {
710  iRRAM_DEBUG1(2,"computation failed, trying normal p-sequence\n");
711  element_step=1;
712  element=4+iRRAM_prec_array[element_step];
713  firsttime=1;
714  } else {
715  iRRAM_DEBUG1(1,"computation of limit_FUNCTION failed totally\n");
716  REITERATE(0);
717  }}
718  }
719  lim.seterror(lim_error);
720  iRRAM_DEBUG2(2,"end of limit_FUNCTION with error %u*2^(%u)\n",
721  lim_error.mantissa, lim_error.exponent);
722  return lim;
723 }
