75 template <class DATA> class iRRAM_cache : public iRRAM_cache_type
76 {
77 public:
78 std::vector<typename get_type<DATA>::type> data;
79 unsigned int current = 0;
80 bool active = false;
81 
82 void put(const DATA& x){
83  if (iRRAM_unlikely(!active)){activate();}
84  data.emplace_back(x);
85  current++;
86 };
87 
88 bool get(DATA& x) noexcept(noexcept(x=x)) {
89  if (current>=data.size())return false;
90  x=data[current++];
91  return true;
92 }
93 
94 void modify(const DATA& x) noexcept(noexcept(const_cast<DATA &>(x)=x)) {
95  data[current-1]=x;
96 }
97 
98 void rewind() noexcept {
99  current=0;
100 };
101 
102 void clear(){
103  data.clear();
104  active=false;
105  current=0;
106 };
107 
108 void activate(){
109  data.clear();
110  active=true;
111  state.cache_active->id[state.max_active++]=this;
