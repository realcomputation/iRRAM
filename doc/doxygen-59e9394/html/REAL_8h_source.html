476 inline REAL & REAL::operator=(const REAL & y)
477 {
478  if (iRRAM_unlikely(value || y.value)) {
479  if (value && y.value) {
480  this->mp_copy(y);
481  return (*this);
482  }
483  if (y.value) {
484  if (state.ACTUAL_STACK.prec_step == 0)
485  this->mp_from_mp(y);
486  else
487  this->mp_copy_init(y);
488  return *this;
489  }
490  dp = y.dp;
491  mp_make_mp();
492  return *this;
493  }
494  dp = y.dp;
495  return *this;

622 inline REAL operator*(const REAL & x, const REAL & y)
623 {
624  if (iRRAM_unlikely(x.value || y.value))
625  return x.mp_conv().mp_multiplication(y.mp_conv());
626  REAL::double_pair z;
627  if (x.dp.lower_pos >= 0 && y.dp.lower_pos >= 0) {
628  z.lower_pos = x.dp.lower_pos * y.dp.lower_pos;
629  z.upper_neg = (-x.dp.upper_neg) * y.dp.upper_neg;
630  } else if (x.dp.upper_neg >= 0 && y.dp.upper_neg >= 0) {
631  z.lower_pos = x.dp.upper_neg * y.dp.upper_neg;
632  z.upper_neg = (-x.dp.lower_pos) * y.dp.lower_pos;
633  } else if (x.dp.upper_neg >= 0 && y.dp.lower_pos >= 0) {
634  z.lower_pos = x.dp.lower_pos * (-y.dp.upper_neg);
635  z.upper_neg = x.dp.upper_neg * y.dp.lower_pos;
636  } else if (x.dp.lower_pos >= 0 && y.dp.upper_neg >= 0) {
637  z.lower_pos = (-x.dp.upper_neg) * y.dp.lower_pos;
638  z.upper_neg = x.dp.lower_pos * y.dp.upper_neg;
639  } else {
640  z.lower_pos = fmin((-x.dp.upper_neg) * y.dp.lower_pos,
641  x.dp.lower_pos * (-y.dp.upper_neg));
642  z.upper_neg = fmin((-x.dp.upper_neg) * y.dp.upper_neg,
643  x.dp.lower_pos * (-y.dp.lower_pos));
644  }
645  return REAL(z);
646 }
