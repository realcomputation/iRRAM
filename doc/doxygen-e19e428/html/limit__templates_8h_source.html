206 /* Requires for Result and ContArgs functions
207  * - int geterror_exp(const C &, const ContArgs &...)
208  * (usually it's enough to specialize the single-argument versions of
209  * geterror_exp(const C &) and geterror_exp(const ContArgs &)... or to
210  * overload geterror(const C &) and geterror(const ContArgs &)...;
211  * then the maximum-norm is used on the vector of all continuous parameters)
212  * - sizetype geterror(const Result &)
213  * - void seterror(Result &, const sizetype &)
214  */
223 template <typename F,typename... ContArgs>
224 auto limit(F f, const ContArgs &... cont_args)
225 -> typename std::enable_if<any_continuous<ContArgs...>::value,
226  decltype(f(0,cont_args...))>::type
227 {
228  using Result = decltype(f(0,cont_args...));
229 
230  limit_computation env;
231 
232  Result lim, limnew;
233  sizetype limnew_error, lim_error;
234  int args_error_exp;
235 
236  int element = env.saved_prec();
237  int element_step = env.saved_step();
238  int firsttime = 2;
239 
240  args_error_exp = geterror_exp(cont_args...);
241 
242  limit_debug("starting general limit_gen1");
243 
244  while (1) {
245  try {
246  iRRAM_DEBUG2(2,"trying to compute general limit_gen1 "
247  "with precicion 2^(%d)...\n", element);
248  limnew = f(element,cont_args...);
249  limnew_error = sizetype_add_power2(geterror(limnew), element);
250  if (firsttime == 2)
251  if (limnew_error.exponent > env.saved_prec(-1)
252  && limnew_error.exponent > args_error_exp - env.saved_prec(-1)) {
253  iRRAM_DEBUG2(2,"computation not precise enough (%d*2^%d), "
254  "trying normal p-sequence\n",
255  limnew_error.mantissa,
256  limnew_error.exponent);
257  element_step = 1;
258  element = 4+iRRAM_prec_array[element_step];
259  firsttime = 1;
260  }
261  if (firsttime != 0 ||
262  sizetype_less(limnew_error,lim_error)) {
263  lim = limnew;
264  lim_error = limnew_error;
265  iRRAM_DEBUG2(2,"getting result with error %d*2^(%d)\n",
266  lim_error.mantissa, lim_error.exponent);
267  } else {
268  iRRAM_DEBUG1(2,"computation successful, but no improvement\n");
269  }
270  firsttime = 0;
271  if (element <= env.saved_prec())
272  break;
273  element_step += 4;
274  element = iRRAM_prec_array[element_step];
275  } catch (const Iteration &it) {
276  if (firsttime == 0) {
277  iRRAM_DEBUG1(2,"computation failed, using best success\n");
278  break;
279  } else if (firsttime == 2) {
280  iRRAM_DEBUG1(2,"computation failed, trying normal p-sequence\n");
281  element_step = 1;
282  element = 4+iRRAM_prec_array[element_step];
283  firsttime = 1;
284  } else {
285  iRRAM_DEBUG1(1,"computation of general limit_gen1 failed totally\n");
286  REITERATE(0);
287  }
288  }
289  }
290  seterror(lim, lim_error);
291  iRRAM_DEBUG2(2,"end of general limit_gen1 with error %d*2^(%d)\n",
292  lim_error.mantissa, lim_error.exponent);
293  return lim;
294 }

331  int* choice,
332  const ARGUMENT&),
333  const ARGUMENT& x)
334 {
335  bool inlimit = state.ACTUAL_STACK.inlimit != 0;
336 
337  limit_computation env;
338 
339  RESULT lim,limnew;
340  sizetype limnew_error,element_error;
341  int choice=0;
342  sizetype lim_error,x_error;
343 
344  int element=env.saved_prec();
345  int element_step=env.saved_step();
346  int firsttime=2;
347 
348  if (!inlimit && !state.thread_data_address->cache_i.get(choice))
349  state.thread_data_address->cache_i.put(choice);
350 
351  x.geterror(x_error);
352 
353  limit_debug("starting limit_mv");
354 
355  while (1) {
356  try {
357  iRRAM_DEBUG2(2,"trying to compute limit_mv with precicion 2^(%d)...\n",element);
358  limnew=f(element,&choice,x);
359  if (!inlimit)
360  state.thread_data_address->cache_i.modify(choice);
361  element_error = sizetype_power2(element);
362  limnew.geterror(limnew_error);
363  limnew_error += element_error;
364  if (firsttime ==2 ) if ( limnew_error.exponent > env.saved_prec(-1)
365  && limnew_error.exponent > x_error.exponent -env.saved_prec(-1)) {
366  iRRAM_DEBUG0(2,{fprintf(stderr,"computation not precise enough (%d*2^%d), trying normal p-sequence\n",
367  limnew_error.mantissa,limnew_error.exponent);});
368  element_step=1;
369  element=4+iRRAM_prec_array[element_step];
370  firsttime=1;
371  }}
372  catch ( Iteration it) {
373  if ( firsttime==0) {
374  iRRAM_DEBUG1(2,"computation failed, using best success\n");
375  break;
376  } else
377  if ( firsttime==2) {
378  iRRAM_DEBUG1(2,"computation failed, trying normal p-sequence\n");
379  element_step=1;
380  element=4+iRRAM_prec_array[element_step];
381  firsttime=1;
382  } else {
383  iRRAM_DEBUG1(1,"computation of limit_gen1 failed totally\n");
384  REITERATE(0);
385  }}
386  if ( firsttime != 0 || sizetype_less(limnew_error,lim_error) ) {
387  lim=limnew;
388  lim_error=limnew_error;
389  iRRAM_DEBUG2(2,"getting result with error %d*2^(%d)\n",
390  lim_error.mantissa, lim_error.exponent);
391  } else {
392  iRRAM_DEBUG1(2,"computation successful, but no improvement\n");
393  }
394  firsttime=0;
395  if (element<=env.saved_prec())break;
396  element_step+=4;
397  element=iRRAM_prec_array[element_step];
398  }
399  lim.seterror(lim_error);
400  iRRAM_DEBUG0(2,{fprintf(stderr,"end of limit_mv with error %d*2^(%d)\n",
401  lim_error.mantissa,lim_error.exponent);});
402  return lim;
403 }

405 template <class ARGUMENT, class DISCRETE, class RESULT>
406 RESULT limit_lip (RESULT (*f)(int,const ARGUMENT&,DISCRETE param),
407  int lip_value,
408  bool (*on_domain)(const ARGUMENT&),
409  const ARGUMENT& x,DISCRETE param)
410 {
411  if ( on_domain(x) != true ) REITERATE(0);
412 
413  limit_computation env;
414 
415  ARGUMENT x_new;
416  RESULT lim;
417  sizetype lim_error,x_error;
418 
419  x_new=x;
420  x_new.geterror(x_error);
421  sizetype_exact(lim_error);
422  x_new.seterror(lim_error);
423 
424  limit_debug("starting limit_lip");
425 
426  while (1) {
427  try{
428  iRRAM_DEBUG2(2,"trying to compute limit_lip with precision %d...\n",state.ACTUAL_STACK.actual_prec);
429  lim=f(env.saved_prec(),x_new,param);
430  lim.geterror(lim_error);
431  if (lim_error.exponent > env.saved_prec()) {
432  env.inc_step(2);
433  limit_debug2("limit_lip too imprecise");
434 
435  } else {
436  iRRAM_DEBUG2(2,"getting result with local error %d*2^(%d)\n",
437  lim_error.mantissa, lim_error.exponent);
438  break;
439  }}
440  catch ( Iteration it){
441  env.inc_step(2);
442  limit_debug2("limit_lip failed");
443  } }
444  lim_error = sizetype_power2(env.saved_prec());
445  lim.adderror(lim_error);
446  lim_error = x_error << lip_value;
447  lim.adderror(lim_error);
448  iRRAM_DEBUG0(2,{lim.geterror(lim_error);
449  fprintf(stderr,"end of limit_lip with error %d*2^(%d)\n",
450  lim_error.mantissa,lim_error.exponent);
451  fprintf(stderr," error of argument: %d*2^(%d)\n",
452  x_error.mantissa,x_error.exponent);});
453  return lim;
454 }

522  int lip,
523  bool (*on_domain)(const CONT_ARGUMENT&, const PARAM& param),
524  const CONT_ARGUMENT& x, const PARAM& param )
