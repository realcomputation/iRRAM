65 static inline int_gmp_type int_gmp_init()
66 {
67  int_gmp_type z;
68  if (gmp_FreeVarCounti>0) {
69  gmp_FreeVarCounti--;
70  z = gmp_FreeVarsi[gmp_FreeVarCounti];
71  }else{
72  z = (int_gmp_type) malloc(sizeof(__mpz_struct));
73  mpz_init(z);
74  }
75  int_gmp_var_count++;
76  return z;
77 }

79 static inline void int_gmp_free(int_gmp_type z)
80 {
81  if (gmp_FreeVarCounti < MaxFreeVars){
82  gmp_FreeVarsi[gmp_FreeVarCounti]=z;
83  gmp_FreeVarCounti++;
84  }else{
85  mpz_clear(z);
86  free(z);
87  }
88  int_gmp_var_count--;
89 }
