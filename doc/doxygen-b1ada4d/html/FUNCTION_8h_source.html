251 template<class PARAM>
252 inline FUNCTION<PARAM,PARAM> polynomial(
253  const std::vector<PARAM> &coeff
254 ) {
255  return from_algorithm(
256  std::function<PARAM(const PARAM &)>(
257  [=](const PARAM &z)
258  {
259  PARAM sum = 0;
260  PARAM factor = 1;
261  for (unsigned int i=0; i<coeff.size(); i++) {
262  sum += factor * coeff[i];
263  factor *= z;
264  }
265  return sum;
266  }
267  )
