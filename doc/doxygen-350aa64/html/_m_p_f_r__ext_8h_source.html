94 iRRAM_STATIC inline void ext_mpfr_remove_trailing_zeroes (mpfr_t x)
95 {
96  unsigned int xn = MPFR_MSW_INDEX(x);
97  unsigned int i;
98  unsigned int j=(xn+1)*BITS_PER_MP_LIMB;
99  mp_limb_t *xp = MPFR_MANT(x);
100 
101  if ( xp[0] != 0 ) return;
102  /* first determine last non-zero limb */
103  for (i=0; i<=xn && xp[i]==0; (i++) ) j-=BITS_PER_MP_LIMB;
104 
105  if (i > xn) return;
106  mpfr_round_prec (x, GMP_RNDN, j);
107 }

296 inline char* ext_mpfr_swritee(ext_mpfr_type z,int w)
297 {
298  char* s=(char*)(malloc(w+1));
299  char* r=&s[2];
300  long e, e_abs,r_len;
301  int i;
302  int flag=0;
303  int sgn = mpfr_sgn (z);
304 
305  mpfr_t z_copy;
306  mpfr_init2(z_copy,mpfr_get_prec(z));
307  mpfr_abs(z_copy,z,iRRAM_mpfr_rounding_mode);
308  mpfr_get_str (r, &e, 10, w-6,z_copy,iRRAM_mpfr_rounding_mode);
309  mpfr_clear(z_copy);
310 
311  /* in case trailing zeroes were discarded, we add them again... */
312  for (i=0;i<w-6;i++) {
313  if ( r[i]==0 ) flag=1 ;
314  if ( flag ) r[i]='0';
315  }
316 
317  /*
318  we want to have an error of less than 0.51ulp, so above
319  we took two additional digits giving an error of at most 0.01ulp
320  (concerning the intended precision).
321  The position of these digits is (w-8,w-7) for the
322  usual 4-decimal exponents and decreases for the larger cases.
323  If the first of the two additional digits is 5 or greater,
324  we round upwards (giving at most additional 0.5ulp)
325  */
326 
327  r_len=e; if (r_len<0) r_len=-r_len;
328  if (r_len<10000){r_len=4;}
329  else if (r_len<100000){r_len=5;}
330  else if (r_len<1000000){r_len=6;}
331  else if (r_len<10000000){r_len=7;}
332  else if (r_len<100000000){r_len=8;}
333  else if (r_len<1000000000){r_len=9;}
334  else {r_len=10;};
335 
336  /*
337  r_len is the number of decimal places of the exponent prior to the rounding
338  */
339 
340 
341  if ( w < r_len+5) w=r_len+5;
342  /*
343  Produce at least one significant digit...
344  */
345 
346  if ( r[w-r_len-4]>='5' ){
347  for (i=w-r_len-5;i>=0;i--){
348  if ( r[i] < '9' ) {
349  r[i]++;
350  break;
351  }
352  r[i]='0';
353  }
354  /*
355  In the rare case that we switch from 0.999...999 (length n) to 1.000...000 (length n+1),
356  we have to correct the exponent and the mantissa, but changing the mantissa is simple:
357  (1) If the (decimal) length of the exponent does not change, result is 0.100...000 (length n)
358  (2) If the (decimal) length of the exponent increases, result is 0.100...000 (length n-1)
359  (3) If the (decimal) length of the exponent decreases, result is 0.100...000 (length n+1)
360  So we simply set the last decimal to '0'. The 0.51ulp accuracy is true in all three cases!
361  */
362  if (i<0){
363  r[0]='1';
364  r[w-r_len-4]='0';
365  e++;
366  }
367  }
368  r[w-r_len-3] = 0;
369  if (sgn== -1) s[0]='-';
370  else s[0]='+';
371  s[1]='.';
372 
373  e_abs=e; if (e_abs<0) e_abs=-e_abs;
374  if (e_abs<10000){e_abs=4;}
375  else if (e_abs<100000){e_abs=5;}
376  else if (e_abs<1000000){e_abs=6;}
377  else if (e_abs<10000000){e_abs=7;}
378  else if (e_abs<100000000){e_abs=8;}
379  else if (e_abs<1000000000){e_abs=9;}
380  else {e_abs=10;};
381  if (e_abs > w-2){sprintf(&s[w-6],"E*****");
382  } else {
383  if (e_abs==4){sprintf(&s[w-6],"E%+05ld",e);}
384  else if (e_abs==5){sprintf(&s[w-7],"E%+06ld",e);}
385  else if (e_abs==6){sprintf(&s[w-8],"E%+07ld",e);}
386  else if (e_abs==7){sprintf(&s[w-9],"E%+08ld",e);}
387  else if (e_abs==8){sprintf(&s[w-10],"E%+09ld",e);}
388  else if (e_abs==9){sprintf(&s[w-11],"E%+010ld",e);}
389  else {sprintf(&s[w-12],"E%+011ld",e);};
390  }
391  return s;
392 }
