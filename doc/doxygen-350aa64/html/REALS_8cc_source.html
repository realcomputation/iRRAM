45 void REAL::mp_make_mp()
46 {
47  if (!std::isfinite(dp.upper_neg) || !std::isfinite(dp.lower_pos))
48  REITERATE(0);
49  if (dp.upper_neg == -dp.lower_pos) {
50  // here we have a point interval...
51  if (!value)
52  MP_init(value);
53  MP_double_to_mp(dp.lower_pos, value);
54  error = sizetype_power2(MP_min);
55  } else {
56  // now we know that it is not a point interval:
57  MP_init(value);
58  MP_double_to_mp(dp.lower_pos, value);
59  MP_type value1;
60  MP_init(value1);
61  MP_double_to_mp(-dp.upper_neg, value1);
62  MP_type value2;
63  MP_init(value2);
64  MP_add(value, value1, value2, -1150);
65  MP_shift(value2, value, -1);
66  MP_clear(value1);
67  MP_clear(value2);
68  // value0 differs from precise result (lower+upper)/2 by at most
69  // 2^{-1200},
70  // which is sufficiently more precise than all bits even of the
71  // smallest double number (2^-1075), so here the error is as
72  // small as reasonable
73  double rwidth = (dp.upper_neg + dp.lower_pos);
74  int e;
75  unsigned m = (unsigned)(1073741824 * frexp(-rwidth, &e)) + 2;
76  // Here, the "+2" accounts for the possible truncation error and
77  // the error on the computation of cvalue. The factor 1073741824
78  // is 2^30
79  // So we have that the interval (dp.lower,dp.upper) is a subset
80  // of the interval (cvalue - m*2^(e+29),cvalue + m*2^(e+29))
81  error = sizetype_normalize({m, e - 29});
82  }
83  MP_getsize(value, vsize);
84 }

86 void REAL::mp_from_mp(const REAL & y)
87 {
88  // d=MP_mp_to_double(y) is required to return a value d such that
89  // there is no other double between d and x.
90  // the conversion should also generate infinities in case of overflow
91  // we don't require a directed rounding here!
92  double center = MP_mp_to_double(y.value);
93  // in consequence, wd will be the negative of an upper bound
94  // for the width of the interval!
95  double wd = ldexp(-double(y.error.mantissa), y.error.exponent);
96  dp.upper_neg = nextafter(-center, -INFINITY) + wd;
97  dp.lower_pos = nextafter(center, -INFINITY) + wd;
98 }

199 std::string swrite(const REAL & x, const int w, const float_form form)
200 {
201  if (!x.value) {
202  REAL y(x);
203  y.mp_make_mp();
204  return swrite(y, w, form);
205  }
206 
207  std::string result;
208  if (state.ACTUAL_STACK.inlimit == 0 &&
209  state.thread_data_address->cache_s.get(result))
210  return result;
211 
212  int width = w;
213  if (width < 9)
214  width = 9;
215  char * erg;
216 
217  switch (form) {
218  case float_form::absolute: {
219  sizetype psize;
220  int p = -10 * (width - 8) / 3;
221  psize = sizetype_power2(p);
222  int s = MP_size(x.value);
223  int mantissa =
224  (int)((s - x.error.exponent - GUARD_BITS) * .30103);
225  if (sizetype_less(psize, x.error) ||
226  (s > p && mantissa + 8 < width)) {
227  iRRAM_DEBUG2(1, "insufficient precision %d*2^(%d) in "
228  "conversion with precision 2^(%d)\n",
229  x.error.mantissa, x.error.exponent, p);
230  REITERATE(p - x.error.exponent);
231  }
232 
233  if (s <= p) {
234  erg = (char *)calloc(width + 1, sizeof(char));
235  strcpy(erg, " 0 ");
236  for (int j = 4; j < width; j++)
237  erg[j] = ' ';
238  erg[width] = 0;
239  } else {
240  erg = MP_swrite(x.value, width);
241  }
242  }
243  case float_form::relative: {
244  int p;
245  {
246  single_valued code;
247  // We want to have an error less than 0.65 ulp, where
248  // 0.51 ulp come from
249  // the conversion to decimal, so we need to know that
250  // the error is at most
251  // 0.15ulp.
252  // At the time, we take the size of x....
253  // It would be better to take the x.size, subtract
254  // x.error and approximately
255  // convert the result to decimal in order to get a
256  // correct approximation
257  // for a lower bound of x. Using this lower bound we can
258  // estimate what
259  // 0.15ulp means in binary.
260  p = size(x) - 2 - 10 * (width - 8) / 3;
261  // We have to prevent that "size" changes the
262  // multi-value-cache here!
263  // Otherwise, any iteration would not use the same
264  // amount of
265  // cache as the first run!
266  }
267  sizetype psize;
268  psize = sizetype_power2(p);
269  int s = MP_size(x.value);
270  int mantissa =
271  (int)((s - x.error.exponent - GUARD_BITS) * .30103);
272  if (sizetype_less(psize, x.error) ||
273  (s > p && mantissa + 8 < width)) {
274  iRRAM_DEBUG2(1, "insufficient precision %d*2^(%d) in "
275  "writing with precision 2^(%d)\n",
276  x.error.mantissa, x.error.exponent, p);
277  REITERATE(p - x.error.exponent);
278  }
279  erg = MP_swrite(x.value, width);
280  }
281  case float_form::show: {
282  int mantissa = (int)(int((MP_size(x.value) - x.error.exponent -
283  GUARD_BITS)) *
284  .30103);
285  if (mantissa < 1) {
286  erg = (char *)calloc(width + 1, sizeof(char));
287  sprintf(erg, " .*E%05.0f",
288  1 + (x.error.exponent + GUARD_BITS) * .30103);
289  for (int j = 9; j < width; j++)
290  erg[j] = ' ';
291  } else {
292  if (mantissa > width - 8)
293  mantissa = width - 8;
294  char * erg2 = MP_swrite(x.value, mantissa + 8);
295  erg = (char *)calloc(width + 1, sizeof(char));
296  strncpy(erg, erg2, mantissa + 8);
297  for (int j = mantissa + 8; j < width; j++)
298  erg[j] = ' ';
299  erg[width] = 0;
300  free(erg2);
301  }
302  }
303  }
304 
305  result = erg;
306  free(erg);
307  if (state.ACTUAL_STACK.inlimit == 0)
308  state.thread_data_address->cache_s.put(result);
309  return result;
310 }

987 int module(REAL f(const REAL&),const REAL& x, int p){
988 // Semantics: If m=module(f,x,p), then
989 // |x-z| <=2^m implies |f(x)-f(z)| <= 2^p
990 
991 // If we are able to approximate f(x) by a DYADIC number d with an error of <=2^{p-1},
992 // then for any z in the argument interval of x, f(z) must differ by at most
993 // 2^{p-1} from d, hence |f(x)-f(z)|<=2^p
994 
995  int result;
996  if ( (state.ACTUAL_STACK.inlimit==0) && state.thread_data_address->cache_i.get(result)) return result;
997 
998  DYADIC d;
999  REAL x_copy=x;
1000 
1001 
1002  sizetype argerror,testerror;
1003 
1004  x_copy.geterror(argerror);
1005  testerror = sizetype_power2(argerror.exponent);
1006  x_copy.adderror(testerror);
1007  {
1008  single_valued code;
1009  d=approx(f(x_copy),p-1);
1010  }
1011 // At this line, we are sure that x_copy (and so also x) is precise enough to allow
1012 // the computation of f(x), even with a slightly increased error of the argument.
1013 
1014 // We now try to find the smallest p_arg such that the evaluation of f(x+- 2^p_arg)
1015 // is possible up to an error of at most 2^{p-1}
1016 
1017 // To do this, we start with p_arg=p.
1018 // If this is successfull, we increase the value of p_arg until the first failure
1019 // It it is not, then we decrease until the first success...
1020  int direction=0,p_arg=p;
1021  bool try_it=true;
1022 
1023  while (try_it) {
1024  testerror = sizetype_power2(p_arg);
1025  x_copy.seterror(argerror);
1026  x_copy.adderror(testerror);
1027  bool fail = false;
1028  if ( iRRAM_unlikely(state.debug > 0 ) ) {
1029  sizetype x_error;
1030  x_copy.geterror(x_error);
1031  iRRAM_DEBUG2(1,"Testing module: 1*2^%d + %d*2^%d\n",p_arg,argerror.mantissa,argerror.exponent);
1032  iRRAM_DEBUG2(1,"argument error: %d*2^%d\n",x_error.mantissa,x_error.exponent);
1033  }
1034  try {
1035  single_valued code;
1036  REAL z=f(x_copy);
1037  if ( iRRAM_unlikely(state.debug > 0 ) ) {
1038  sizetype z_error;
1039  z.geterror(z_error);
1040  iRRAM_DEBUG2(1,"Module yields result %d*2^%d\n",z_error.mantissa,z_error.exponent);
1041  }
1042  d=approx(z,p-1);
1043  }
1044  catch ( Iteration it) { fail=true; }
1045  switch ( direction ) {
1046  case 0:;
1047  if ( fail ) direction=-1; else direction=1;
1048  p_arg+=direction;
1049  break;
1050  case 1:;
1051  if ( fail ) { try_it=false; p_arg -=direction; }
1052  else { p_arg += direction; };
1053  break;
1054  case -1:;
1055  if ( fail ) { p_arg +=direction; }
1056  else { try_it=false; };
1057  break;
1058  }
1059  }
1060  iRRAM_DEBUG2(1,"Modules resulting in p_arg=%d\n",p_arg);
1061 
1062  testerror = sizetype_power2(p_arg);
1063  argerror += testerror;
1064 
1065  while (argerror.mantissa>1) {
1066  argerror.mantissa=argerror.mantissa>>1;
1067  argerror.exponent+=1;
1068  }
1069 
1070  result=argerror.exponent;
1071  if ( state.ACTUAL_STACK.inlimit==0 ) state.thread_data_address->cache_i.put(result);
1072  return result;
1073 
1074 }
