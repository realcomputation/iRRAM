153 template <class PARAM>
154 static orstream & iRRAM_out(orstream * s, const PARAM & x)
155 {
156  if ((state.ACTUAL_STACK.inlimit > 0) && s->_respect_iteration) {
157  iRRAM_DEBUG1(2, "illegal output in continuous section!\n");
158  return *s;
159  }
160  if ((state.ACTUAL_STACK.inlimit == 0) && s->_respect_iteration) {
161  single_valued code;
162  if (++state.requests > state.outputs) {
163  *s->target << x;
164  state.outputs++;
165  }
166  } else {
167  *s->target << x;
168  }
169  return *s;
170 }
