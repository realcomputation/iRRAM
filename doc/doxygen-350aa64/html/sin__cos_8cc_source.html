18 static REAL sin_taylor(int prec, const REAL & x)
19 {
20  // We compute sin(x) to a precision of 2^prec using the Taylor
21  // expansion. The evaluation of the expansion is improved by grouping
22  // wd values together.
23  // The argument x will be always smaller than 1,
24  // usually it will be very small...
25 
26  // All computations in this routine are exact,
27  // but the truncation error must be taken into account.
28 
29  int sx = upperbound(x);
30  if (prec >= sx)
31  return 0;
32 
33  int wd;
34  if (prec > -100) wd = 1;
35  else if (prec > -200) wd = 1;
36  else if (prec > -400) wd = 2;
37  else if (prec > -800) wd = 2;
38  else if (prec > -1600) wd = 2;
39  else if (prec > -3200) wd = 3;
40  else if (prec > -6400) wd = 4;
41  else if (prec > -12000) wd = 5;
42  else if (prec > -24000) wd = 8;
43  else if (prec > -48000) wd = 12;
44  else if (prec > -100000) wd = 18;
45  else if (prec > -200000) wd = 26;
46  else wd = int(std::sqrt(double(-prec) / 18));
47 
48  REAL x2 = square(x);
49  std::vector<REAL> xpow(wd);
50  REAL xf = power(x2, wd), z = x;
51  xpow[0] = x;
52  for (int i = 1; i < wd; i++) {
53  xpow[i] = xpow[i - 1] * x2 / (-2 * i * (2 * i + 1));
54  z += xpow[i];
55  }
56  REAL xp = 1, e = z;
57  INTEGER f = 1;
58  int i = 1;
59  int t = upperbound(xf), t1 = t;
60  while (!bound(e, prec) || !bound(e, -1)) {
61  xp *= xf;
62  e = 0;
63  for (int j = 0; j < wd; j++) {
64  f = 1;
65  for (int k = i; k < i + wd; k++)
66  f *= (-2 * k * (2 * k + 1));
67  xpow[j] = scale(xpow[j], t) / REAL(f);
68  i = i + 1;
69  e += xpow[j];
70  }
71  e *= scale(xp, -t1);
72  t1 = t1 + t;
73  z += e;
74  }
75  // printf("%d %d %d\n",i,prec,upperbound(x));
76  return z;
77 }

125 static REAL sin_range_red1(int prec, const REAL & x)
126 {
127  // This function computes an approximation to sin(x) for arbitray x.
128 
129  // To achieve this, we do a range reduction of x to the interval
130  // [0-eps,pi/4+eps] where eps<=0.01. This reduction is done precisely,
131  // so that is sufficient to use identities on sine
132  // for the actual computation of sin(x).
133 
134  // With the help of a further algorithm working fast on this area,
135  // we then do the actual computation of sin(x).
136  // The precision is only needed as a hint how to do this quite fast!
137 
138  // As |sin(x)|<=1, the number 0 is a valid approximation to cos(x)
139  // with precision 2^{-0}=1
140  if (prec >= 0)
141  return 0;
142  // If the required precision is higher, we actually have to compute
143  // sin(x)
144  // In this case, the result is not an approximation but precise, indeed!
145 
146  int x_int;
147  REAL x_red;
148  bool sin_neg;
149  // For larger numbers, we increase the working precision a bit:
150  {
151  stiff code(!bound(x, 2) ? +1 : 0);
152 
153  // We reduce x to -2pi..2pi by taking the value modulo 2pi
154  x_red = modulo(x, 2 * pi());
155  // We take an integer approximation x_int of 100*x_red,
156  // which must be between -629..629
157  // This allows us to easily find a reasonable reduction interval
158  x_int = round(x_red * 100);
159  // We have |x_int:100-x_red| <= 0.01
160 
161  // In case x_int is negative, we shift by 2pi(),
162  // so we still have sin(x)=sin(x_red)
163  if (x_int < 0) {
164  x_int = x_int + 628;
165  x_red = x_red + 2 * pi();
166  }
167  // We now have /x_int:100 - x_red/ < 0.02, 0<=x_int<=629 and
168  // -0.01 <= x_red <= 2pi
169 
170  // In case x_int > 314, we reduce using sin(x)=-sin(x-pi)
171  // so we still have sin(x)=sin(x_red)*sin_sign;
172  sin_neg = false;
173  if (x_int > 314) {
174  x_int = x_int - 314;
175  x_red = x_red - pi();
176  sin_neg = true;
177  }
178  // We now have /x_int:100 - x_red/ < 0.03 and
179  // -0.01 <= x_red <= pi+0.01
180 
181  // In case x_int > 157, we reduce using sin(x)=sin(pi-x),
182  // so still sin(x)=sin(x_red)*sin_sign;
183  if (x_int > 157) {
184  x_int = 157 - x_int;
185  x_red = pi() - x_red;
186  }
187  // We now have /x_int:100 - x_red/ < 0.04 and
188  // -0.02 <= x_red <= pi/2+0.02
189  }
190 
191  // Now we do an exact(!) computation of sin(x).
192  // The procedure uses the argument prec just as an hint for
193  // optimization!
194  REAL sin_abs;
195  if (x_int > 80)
196  sin_abs = sqrt(1 -
197  square(sin_range_red3(prec, x_red - pi() / 2)));
198  else
199  sin_abs = sin_range_red3(prec, x_red);
200  return sin_neg ? -sin_abs : sin_abs;
201 }
