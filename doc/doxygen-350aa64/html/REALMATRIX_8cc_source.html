131 REALMATRIX solve
132  (REALMATRIX& lside,
133  REALMATRIX& rside,
134  int use_pivot) {
135  single_valued code;
136  REALMATRIX result(rside.maxrow,rside.maxcolumn);
137  REAL faktor;
138  int testsize,pivotsize,pivot,pivotsave;
139  std::vector<int> perm (lside.maxrow);
140 
141  if ( lside.maxrow != rside.maxrow) {
142  fprintf(stderr,"Error, incompatible real matrices\n");
143  exit(1);
144  }
145  for (unsigned int i=0; i<lside.maxrow;i++) perm[i]=i;
146  for (unsigned int i=0; i < lside.maxrow; i++) {
147  pivot=i;
148  if (use_pivot) {
149  pivotsize=upperbound(ELEMENT(lside,perm[i],i));
150  for (unsigned int j=i+1; j < lside.maxcolumn; j++) {
151  testsize=upperbound(ELEMENT(lside,perm[j],i));
152  if (testsize>pivotsize) {
153  pivot=j;
154  pivotsize=testsize;
155  }
156  else if ( testsize < pivotsize && use_pivot == 1) break ;
157  }
158  pivotsave=perm[pivot];
159  perm[pivot]=perm[i];
160  perm[i]=pivotsave;
161  }
162  for (unsigned int j=i+1; j < lside.maxcolumn; j++) {
163  faktor=ELEMENT(lside,perm[j],i)/ELEMENT(lside,perm[i],i);
164  for (unsigned int k=0; k< rside.maxcolumn; k++)
165  ELEMENT(rside,perm[j],k)=ELEMENT(rside,perm[j],k)
166  -faktor*ELEMENT(rside,perm[i],k);
167  for (unsigned int l=i; l< lside.maxcolumn; l++)
168  ELEMENT(lside,perm[j],l)=ELEMENT(lside,perm[j],l)
169  -faktor*ELEMENT(lside,perm[i],l);
170  }
171  }
172 
173  for (int i=rside.maxrow-1; i >= 0; i--) {
174  faktor=1/ELEMENT(lside,perm[i],i);
175  for (unsigned int k=0; k< rside.maxcolumn; k++) {
176  ELEMENT(result,i,k)=ELEMENT(rside,perm[i],k)*faktor;
177  for (int j=0; j < i; j++)
178  ELEMENT(rside,perm[j],k)=ELEMENT(rside,perm[j],k)
179  -ELEMENT(lside,perm[j],i)* ELEMENT(result,i,k);
180  } }
181  return result;
182 }

300 static REALMATRIX expmatrix_approx (int prec, const REALMATRIX& x){
301  if (rows(x)!=columns(x)) {
302  fprintf(stderr,"Exponentiation of non-quadratic real matrix [%d,%d]\n",
303  columns(x),rows(x));
304  exit(1);
305  }
306  if ( prec >= 2 ) return eye(rows(x));
307  REALMATRIX y=x,z=y+eye(rows(x));
308  int i=2;
309  while ( !bound(y,prec) ) {
310  y=y*x/i;
311  z=z+y;
312  i+=1;
313  }
314  return z;
315 }

334 REALMATRIX steady_state (const REALMATRIX& x) {
335  single_valued code;
336  REALMATRIX result(1,rows(x));
337  REALMATRIX chain=x;
338  REAL factor,outrate;
339  if ( rows(x) != columns(x)) {
340  fprintf(stderr,"Argument is not a square matrix\n");
341  exit(1);
342  }
343  for (unsigned int N=0; N < rows(x); N++) chain(N,N)=chain(N,N)-1;
344 
345  for (unsigned int N=0; N < rows(x)-1; N++) {
346  for (unsigned int i=N+1; i < columns(x); i++) {
347  factor=chain(i,N)/chain(N,N);
348  for (unsigned int j=N+1; j< columns(x); j++)
349  chain(i,j)=chain(i,j)
350  -factor*chain(N,j);
351  } }
352 
353  result(0,rows(x)-1)=1;
354  for (int N=rows(x)-2; N >= 0; N--) {
355  outrate=0;
356  for (int j=rows(x)-1; j>N;j--) outrate=outrate
357  +result(0,j)*chain(j,N);
358  result(0,N)=outrate/(outrate-chain(N,N));
359  factor=-chain(N,N)/(outrate-chain(N,N));
360  for (int j=rows(x)-1; j>N;j--)
361  result(0,j)=result(0,j)*factor;
362  }
363  return result;
364 }
