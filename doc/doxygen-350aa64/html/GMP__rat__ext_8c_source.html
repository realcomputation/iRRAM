46 void rat_gmp_add_si(const rat_gmp_type z1, int z2, rat_gmp_type z)
47 {
48  int_gmp_type src_num = mpq_numref(z1);
49  int_gmp_type src_den = mpq_denref(z1);
50  int_gmp_type tgt_num = mpq_numref(z);
51  int_gmp_type tgt_den = mpq_denref(z);
52  int_gmp_type temp = int_gmp_init();
53 
54  mpz_set(tgt_den, src_den);
55  mpz_mul_si(temp, src_den, z2);
56  mpz_add(tgt_num, temp, src_num);
57 
58  mpq_canonicalize(z);
59  int_gmp_free(temp);
60 }
